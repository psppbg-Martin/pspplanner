// PSP Workforce Planner — Settings + Vacation Calendar API
// Deploy: Extensions > Apps Script > Deploy > Web App
// Execute as: Me  |  Who has access: Anyone
// After deploying, paste the Web App URL into Data Source above.

var MONTH_NAMES = ['january','february','march','april','may','june',
                   'july','august','september','october','november','december'];
var SETTINGS_KEY = 'psp_settings';

function doGet(e) {
  var params = (e && e.parameter) ? e.parameter : {};
  var cb     = params.callback || '';
  var mode   = params.mode    || '';

  var respond = function(obj) {
    var json = JSON.stringify(obj);
    var out  = cb ? cb + '(' + json + ')' : json;
    var mime = cb ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON;
    return ContentService.createTextOutput(out).setMimeType(mime);
  };

  // ── Save a settings chunk (GET-based, no CORS issues) ─────────────────────
  if (mode === 'saveChunk') {
    var chunk = parseInt(params.chunk || '0');
    var total = parseInt(params.total || '1');
    var data  = params.data || '';
    var props = PropertiesService.getScriptProperties();

    if (chunk === 0) {
      // First chunk — clear any previous chunks
      for (var i = 0; i < 20; i++) props.deleteProperty(SETTINGS_KEY + '_' + i);
      props.setProperty(SETTINGS_KEY + '_total', String(total));
    }
    props.setProperty(SETTINGS_KEY + '_' + chunk, data);

    // If all chunks received, assemble and save
    if (chunk === total - 1) {
      var assembled = '';
      for (var j = 0; j < total; j++) {
        assembled += (props.getProperty(SETTINGS_KEY + '_' + j) || '');
      }
      props.setProperty(SETTINGS_KEY, assembled);
    }
    return respond({ ok: true, chunk: chunk, total: total });
  }

  // ── Load settings ──────────────────────────────────────────────────────────
  if (mode === 'loadSettings') {
    var saved = PropertiesService.getScriptProperties().getProperty(SETTINGS_KEY);
    return respond({ ok: !!saved, data: saved || null });
  }

  // ── Debug: list sheet names ────────────────────────────────────────────────
  if (mode === 'debug') {
    var ss2  = SpreadsheetApp.getActiveSpreadsheet();
    var info = { spreadsheetName: ss2.getName(), sheets: [] };
    ss2.getSheets().forEach(function(sh) {
      var lastR = Math.min(6, sh.getLastRow());
      var lastC = Math.min(8, sh.getLastColumn());
      var rows  = lastR > 0 && lastC > 0 ? sh.getRange(1,1,lastR,lastC).getValues() : [];
      info.sheets.push({ name: sh.getName(), sampleRows: rows });
    });
    return respond(info);
  }

  // ── Main: read vacation absences from spreadsheet ─────────────────────────
  var ss     = SpreadsheetApp.getActiveSpreadsheet();
  var result = {};
  ss.getSheets().forEach(function(sheet) {
    var sheetName = sheet.getName().trim();
    var mIdx = MONTH_NAMES.indexOf(sheetName.toLowerCase());
    if (mIdx === -1) { var n = parseInt(sheetName); if (n >= 1 && n <= 12) mIdx = n - 1; }
    if (mIdx === -1) return;
    var lastRow = sheet.getLastRow(), lastCol = sheet.getLastColumn();
    if (lastRow < 1 || lastCol < 1) return;
    var data = sheet.getRange(1, 1, lastRow, Math.max(lastCol, 8)).getValues();
    function isDate(v) { return v instanceof Date && !isNaN(v.getTime()); }
    function fmt(d) {
      return d.getFullYear() + '-' +
        String(d.getMonth()+1).padStart(2,'0') + '-' +
        String(d.getDate()).padStart(2,'0');
    }
    function isWeekend(d) { var w = d.getDay(); return w === 0 || w === 6; }
    var i = 0;
    while (i < data.length) {
      var row = data[i], dh = [];
      for (var c = 1; c <= 7; c++) dh.push(isDate(row[c]) ? row[c] : null);
      if (!dh.some(function(d) { return d !== null; })) { i++; continue; }
      var j = i + 1;
      while (j < data.length) {
        var nr = data[j], nd = false;
        for (var c2 = 1; c2 <= 7; c2++) { if (isDate(nr[c2])) { nd = true; break; } }
        if (nd) break;
        for (var c = 1; c <= 7; c++) {
          var d = dh[c-1]; if (!d || isWeekend(d)) continue;
          var cell = nr[c]; if (!cell) continue;
          var val = String(cell).trim();
          if (!val || val === '0' || val.toLowerCase() === 'false') continue;
          if (val.toUpperCase() === 'HOLIDAY' || val.toUpperCase() === 'HOLIDAY OBSERVED') continue;
          var ds = fmt(d);
          val.split(/  +/).forEach(function(p) {
            var nm = p.trim(); if (!nm || nm === '-') return;
            if (!result[ds]) result[ds] = [];
            if (result[ds].indexOf(nm) === -1) result[ds].push(nm);
          });
        }
        j++;
      }
      i = j;
    }
  });
  return respond(result);
}
